# 开始
* 安装依赖 pip install -r requirements.txt
* 直接执行在Run目录下的main.py文件
* 默认flask的地址为172.0.0.1:5010
* 支持Python3.x


# config.py配置文件
* redis_host = '127.0.0.1' #redis 连接地址
* redis_port = 6379 #redis 端口
* flask_host = '0.0.0.0' # flask 地址
* flask_port = 5010 # flask 端口
* db = 0 #数据库
* name = 'proxy' #数据库表名
